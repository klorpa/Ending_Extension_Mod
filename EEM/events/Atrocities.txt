    desc = "EVTDESC23111"
    picture = "war_crimes"
    is_triggered_only = yes

    option = {
        name = "This is terrible!"
        relation = {
            who = FROM
            value = -50
        }
        any_owned = {
            limit = {
                controlled_by = FROM
                NOT = { province_control_days = 365 }
                NOT = { has_province_modifier = the_massacre }
            }
            add_province_modifier = {
                name = the_massacre
                duration = 90
            }
        }

        NWO = {
            change_variable = {
                which = war_crime_counter
                value = 1
            }
        }
    }
